% Load USPS Handwritten Digits dataset
load('usps_all.mat');

% Check the variable names in the dataset
whos

% Initialize a figure
figure;

% Iterate over each digit from 1 to 9 and 0
for i = 1:10
    % Select the first image of the digit i
    img = data(:,i,i); % Selecting first image of the i-th set
    % Plot the image
    subplot(2, 5, i);
    imshow(reshape(img, [16 16]));
    title(mod(i-1, 10));
end

% Save the figure as a PDF
saveas(gcf, 'Q1_USPS_Digits.pdf');
